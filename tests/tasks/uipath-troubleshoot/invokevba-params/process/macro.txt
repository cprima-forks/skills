Sub PostRow(amount As Double, vendor As String)
    Dim ws As Worksheet
    Set ws = ActiveWorkbook.Sheets("Ledger")
    Dim nextRow As Long
    nextRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row + 1
    ws.Cells(nextRow, 1).Value = Now
    ws.Cells(nextRow, 2).Value = vendor
    ws.Cells(nextRow, 3).Value = amount
    ActiveWorkbook.Save
End Sub
